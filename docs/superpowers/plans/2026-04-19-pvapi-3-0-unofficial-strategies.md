# pvapi 3.0 — unofficial strategies (plan 7) implementation

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Let users create portfolios from arbitrary GitHub strategy repositories. Every portfolio carries `strategy_clone_url` + `strategy_describe_json`. The `strategies` table becomes a build cache for officials only; unofficials go ephemeral every backtest.

**Architecture:** Unified backtest resolver keyed on `(clone_url, strategy_ver)`. Officials populate the cache via the plan-3 sync goroutine and hit warm. Unofficials never populate — they always fall through to `strategy.EphemeralBuild`, which clones to a tempdir, builds, runs, and cleans up via a closure deferred by the orchestrator. A new `GET /strategies/describe` endpoint performs the same clone+build+describe and returns the describe JSON so the UI can render parameter forms; it persists nothing.

**Tech Stack:** Go 1.23+, Fiber v3, pgx v5, modernc SQLite, Ginkgo/Gomega, oapi-codegen, viper, zerolog. Reuses `strategy/testdata/fake-strategy-src` for every build-touching test.

**Spec:** `docs/superpowers/specs/2026-04-19-pvapi-3-0-unofficial-strategies.md`.

```go
// Package strategy — ephemeral build support for unofficial strategies.
package strategy

import (
    "errors"
    "regexp"
)

var (
    ErrInvalidCloneURL = errors.New("cloneUrl must be an https GitHub URL of the form https://github.com/<owner>/<repo>(.git)?")

    cloneURLRe = regexp.MustCompile(`^https://github\.com/[A-Za-z0-9_.-]+/[A-Za-z0-9_.-]+(\.git)?$`)
)

// ValidateCloneURL returns nil if cloneURL matches the allowlist, else
// ErrInvalidCloneURL.
func ValidateCloneURL(cloneURL string) error {
    if cloneURL == "" || !cloneURLRe.MatchString(cloneURL) {
        return ErrInvalidCloneURL
    }
    return nil
}
```

```go
func copyTree(src, dst string) error {
    return filepath.Walk(src, func(p string, info os.FileInfo, err error) error {
        if err != nil {
            return err
        }
        rel, err := filepath.Rel(src, p)
        if err != nil {
            return err
        }
        target := filepath.Join(dst, rel)
        if info.IsDir() {
            return os.MkdirAll(target, 0o755)
        }
        data, err := os.ReadFile(p)
        if err != nil {
            return err
        }
        return os.WriteFile(target, data, info.Mode())
    })
}
```

```go
import (
    "bytes"
    "context"
    "fmt"
    "os"
    "os/exec"
    "path/filepath"
    "time"
)

// EphemeralOptions configures an ephemeral clone+build.
type EphemeralOptions struct {
    CloneURL          string        // required
    Ver               string        // optional; empty = default branch HEAD
    Dir               string        // parent dir for mkdtemp; empty = os.TempDir()
    Timeout           time.Duration // 0 = 60s
    SkipURLValidation bool          // tests may relax the allowlist
}

const defaultEphemeralTimeout = 60 * time.Second

// EphemeralBuild clones CloneURL into a fresh tempdir under Dir, runs
// `go build .`, and returns the absolute binary path plus a cleanup
// closure that removes the tempdir. The closure is safe to call
// multiple times and is the caller's responsibility to invoke (the
// caller typically `defer`s it).
//
// On any error before returning successfully, EphemeralBuild removes
// the tempdir itself and returns ("", nil, err).
func EphemeralBuild(ctx context.Context, opts EphemeralOptions) (string, func(), error) {
    if !opts.SkipURLValidation {
        if err := ValidateCloneURL(opts.CloneURL); err != nil {
            return "", nil, err
        }
    }
    timeout := opts.Timeout
    if timeout <= 0 {
        timeout = defaultEphemeralTimeout
    }
    parent := opts.Dir
    if parent == "" {
        parent = os.TempDir()
    }
    if err := os.MkdirAll(parent, 0o750); err != nil {
        return "", nil, fmt.Errorf("mkdir ephemeral parent: %w", err)
    }
    dir, err := os.MkdirTemp(parent, "build-*")
    if err != nil {
        return "", nil, fmt.Errorf("mkdtemp: %w", err)
    }

    var removed bool
    cleanup := func() {
        if removed {
            return
        }
        removed = true
        _ = os.RemoveAll(dir)
    }

    tctx, cancel := context.WithTimeout(ctx, timeout)
    defer cancel()

    cloneArgs := []string{"clone", "--depth=1"}
    if opts.Ver != "" {
        cloneArgs = append(cloneArgs, "--branch", opts.Ver)
    }
    cloneArgs = append(cloneArgs, opts.CloneURL, dir)
    var cloneBuf bytes.Buffer
    cloneCmd := exec.CommandContext(tctx, "git", cloneArgs...) //nolint:gosec // cloneURL validated above
    cloneCmd.Stdout = &cloneBuf
    cloneCmd.Stderr = &cloneBuf
    if err := cloneCmd.Run(); err != nil {
        cleanup()
        return "", nil, fmt.Errorf("git clone %s: %w\n%s", opts.CloneURL, err, cloneBuf.String())
    }

    binPath := filepath.Join(dir, "strategy.bin")
    var buildBuf bytes.Buffer
    buildCmd := exec.CommandContext(tctx, "go", "build", "-o", binPath, ".") //nolint:gosec // binPath internal
    buildCmd.Dir = dir
    buildCmd.Stdout = &buildBuf
    buildCmd.Stderr = &buildBuf
    if err := buildCmd.Run(); err != nil {
        cleanup()
        return "", nil, fmt.Errorf("go build: %w\n%s", err, buildBuf.String())
    }

    return binPath, cleanup, nil
}
```
